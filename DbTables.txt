drop table jDevs_skills;

drop table JuniorDevelopers;
drop table Skills;
drop table Mentors;

TRUNCATE TABLE Mentors;

create table Mentors(
	ID varchar(50) primary key,
	FirstName varchar(50),
	LastName varchar(50),
	BDate timestamptz,
	Sex int,
	Salary int,
	HDate timestamptz
)

create table Skills (
	ID varchar(50) primary key,
	Name varchar(50),
	KMentor varchar(50),
	foreign key (KMentor) references Mentors (ID)
)

create table JuniorDevelopers (
	ID varchar(50) primary key,
	FirstName varchar(50),
	LastName varchar(50),
	BDate timestamptz,
	Sex int,
	Salary varchar(50),
	HDate timestamptz,
	KSkill varchar(50),
	foreign key (KSkill) references Skills (ID)
)

create table Departments (
	ID varchar(50) primary key,
	Name varchar(50),
	KMentor varchar(50),
	foreign key (KMentor) references Mentors (ID)
)


create table jDevs_skills(
	KjDev varchar(50),
	KSkill varchar(50),
	foreign key (KjDev) references JuniorDevelopers (ID),
	foreign key (KSkill) references Skills (ID)
)
